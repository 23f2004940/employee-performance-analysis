import pandas as pd
import matplotlib.pyplot as plt

data = {
    "EmployeeID": range(1, 101),
    "Department": ["Finance"]*15 + ["HR"]*20 + ["IT"]*25 + ["Sales"]*40
}
df = pd.DataFrame(data)

finance_count = (df["Department"] == "Finance").sum()
print("Number of employees in Finance department:", finance_count)

plt.figure(figsize=(6,4))
df["Department"].value_counts().plot(kind="bar", color="skyblue", edgecolor="black")
plt.title("Department Distribution")
plt.xlabel("Department")
plt.ylabel("Count")
plt.tight_layout()
plt.savefig("hist.png")
